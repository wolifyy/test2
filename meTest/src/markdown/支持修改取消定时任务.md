以下是一个基本的实现示例：

```java
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.TimeUnit;

public class TaskScheduler {
    private Map<String, ScheduledTask> tasks;
    private ScheduledExecutorService executor;

    public TaskScheduler() {
        tasks = new HashMap<>();
        executor = Executors.newScheduledThreadPool(20);
    }

    public void scheduleTask(String taskId, long initialDelay, long period, TimeUnit unit, Runnable task) {
        ScheduledTask scheduledTask = new ScheduledTask(taskId, initialDelay, period, unit, task);
        tasks.put(taskId, scheduledTask);
        executor.scheduleAtFixedRate(scheduledTask, initialDelay, period, unit);
    }

    public void cancelTask(String taskId) {
        ScheduledTask scheduledTask = tasks.get(taskId);
        if (scheduledTask != null) {
            scheduledTask.cancel();
            tasks.remove(taskId);
        }
    }

    public void modifyTask(String taskId, long initialDelay, long period, TimeUnit unit) {
        ScheduledTask scheduledTask = tasks.get(taskId);
        if (scheduledTask != null) {
            scheduledTask.modify(initialDelay, period, unit);
        }
    }

    private class ScheduledTask implements Runnable {
        private String taskId;
        private long initialDelay;
        private long period;
        private TimeUnit unit;
        private Runnable task;
        private volatile boolean cancelled;

        public ScheduledTask(String taskId, long initialDelay, long period, TimeUnit unit, Runnable task) {
            this.taskId = taskId;
            this.initialDelay = initialDelay;
            this.period = period;
            this.unit = unit;
            this.task = task;
            this.cancelled = false;
        }

        public void cancel() {
            cancelled = true;
        }

        public void modify(long initialDelay, long period, TimeUnit unit) {
            this.initialDelay = initialDelay;
            this.period = period;
            this.unit = unit;
        }

        @Override
        public void run() {
            if (!cancelled) {
                task.run();
            }
        }
    }
}
```

使用示例：

```java
TaskScheduler scheduler = new TaskScheduler();

// 创建任务并定时执行
scheduler.scheduleTask("task1", 0, 1, TimeUnit.MINUTES, () -> {
    System.out.println("Task 1 executed");
});

// 修改任务的执行周期
scheduler.modifyTask("task1", 0, 30, TimeUnit.SECONDS);

// 取消任务
scheduler.cancelTask("task1");
```
